### Sub-Schema Generation Examples

*** Example 1 ***
### Database: Sales
### Database Schema

CREATE TABLE Customers (
    CustomerID PK,
    FirstName,
    MiddleInitial,
    LastName
)

CREATE TABLE Employees (
    EmployeeID PK,
    FirstName,
    MiddleInitial,
    LastName
)

CREATE TABLE Products (
    ProductID PK,
    Name,
    Price
)

CREATE TABLE Sales (
    SalesID PK,
    SalesPersonID,
    CustomerID,
    ProductID,
    Quantity
)

### All possible sub-schemas for the Sales database:
{{
    "sub_schema_1": [Customers],
    "sub_schema_2": [Employees],
    "sub_schema_3": [Products],
    "sub_schema_4": [Sales],
    "sub_schema_5": [Customers, Sales],
    "sub_schema_6": [Employees, Sales],
    "sub_schema_7": [Products, Sales],
    "sub_schema_8": [Customers, Sales, Employees],
    "sub_schema_9": [Customers, Sales, Products],
    "sub_schema_10": [Employees, Sales, Products],
    "sub_schema_11": [Employees, Sales, Products],
    "sub_schema_12": [Customers, Sales, Employees, Products]
}}

*** Example 2 ***
### Database: authors
### Database Schema

CREATE TABLE Author (
    Id PK,
    Name,
    Affiliation, 
)

CREATE TABLE Conference (
    Id PK,
    ShortName,
    FullName,
    HomePage,
)

CREATE TABLE Journal (
    Id PK,
    ShortName,
    FullName,
    HomePage
)

CREATE TABLE Paper (
    Id PK,
    Title,
    Year,
    ConferenceId,
    JournalId,
    Keyword
)

CREATE TABLE PaperAuthor (
    PaperId,
    AuthorId,
    Name,
    Affiliation
)

### All possible sub-schemas for the authors database:
{{
    "sub_schema_1": [Author],
    "sub_schema_2": [Conference],
    "sub_schema_3": [Journal],
    "sub_schema_4": [Paper],
    "sub_schema_5": [PaperAuthor],
    "sub_schema_6": [Author, PaperAuthor],
    "sub_schema_7": [Conference, Paper],
    "sub_schema_8": [Journal, Paper],
    "sub_schema_9": [Paper, PaperAuthor],
    "sub_schema_10": [Author, PaperAuthor, Paper],
    "sub_schema_11": [Conference, Paper, PaperAuthor],
    "sub_schema_12": [Journal, Paper, PaperAuthor],
    "sub_schema_13": [Author, PaperAuthor, Paper, Conference],
    "sub_schema_14": [Author, PaperAuthor, Paper, Journal],
    "sub_schema_15": [Author, PaperAuthor, Paper, Conference, Journal],

}}

*** Example 3 ***
### Database: law_episode
### Database Schema

CREATE TABLE Episode (
    Id PK,
    series,
    season,
    episode,
    number_in_series,
    title,
    summary,
    air_date,
    episode_image,
    rating,
    votes,
)

CREATE TABLE Keyword (
    episode_id PK,  
    keyword PK
)

CREATE TABLE Person (
    person_id PK,
    name,
    birthdate,
    birth_name,
    birth_place,
    birth_region,
    birth_country,
    height_meters,
    nickname
)

CREATE TABLE Award (
    award_id PK,
    organization,
    year,
    award_category,
    award,
    series,
    episode_id, 
    person_id,  
    role,
    result
)

CREATE TABLE Credit (
    episode_id PK,  
    person_id PK,  
    category,
    role,
    credited
)

CREATE TABLE Vote (
    episode_id PK,  (FK to Episode.Id)
    stars,
    votes,
    percent
)
